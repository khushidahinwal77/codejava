import java.awt.*;
        class LabelFlowLayout extends Frame {
          Label lbl = new Label("Welcome to AWT!");
          LabelFlowLayout(){
            setLayout(new FlowLayout());
            add(lbl);
            setSize(300,200);
            setVisible(true);
          }
          public static void main(String ap[]){
            new LabelFlowLayout();
          }
          }
