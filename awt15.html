import java.awt.*;
        class ListCustomLayout extends Frame {
          List lst = new List();
          ListCustomLayout(){
            lst.add("Item 1");
            lst.add("Item 2");
            lst.add("Item 3");
            setLayout(new GridLayout(1, 1));
            add(lst);
            setSize(300,200);
            setVisible(true);
          }
          public static void main(String ap[]){
            new ListCustomLayout();
          }
          }
