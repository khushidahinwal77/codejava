import java.awt.*;
        class CheckboxGroupFlowLayout extends Frame {
          CheckboxGroup group = new CheckboxGroup();
          Checkbox cb1 = new Checkbox("Option 1", group, false);
          Checkbox cb2 = new Checkbox("Option 2", group, false);
          Checkbox cb3 = new Checkbox("Option 3", group, false);
          CheckboxGroupFlowLayout(){
            setLayout(new FlowLayout());
            add(cb1);
            add(cb2);
            add(cb3);
            setSize(300,200);
            setVisible(true);
          }
          public static void main(String ap[]){
            new CheckboxGroupFlowLayout();
          }
          }
