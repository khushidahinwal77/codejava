import java.awt.*;
        import java.awt.event.*;
        class CloseFrameExample extends Frame {
          CloseFrameExample(){
            setSize(400,200);
            setVisible(true);
            addWindowListener(new WindowAdapter(){
              public void windowClosing(WindowEvent we){
                System.exit(0);
              }
              });
          }
          public static void main(String ap[]){
            new CloseFrameExample();
          }
          }
