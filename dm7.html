import java.awt.*;
    import java.awt.event.*;
    class CheckBoxExample extends Frame {
      Checkbox checkbox = new Checkbox("Accept Terms");
      Button btn = new Button("Check Status");
      CheckBoxExample(){
        add(checkbox);
        add(btn);
        setLayout(new FlowLayout());
        setSize(300,200);
        setVisible(true);
        btn.addActionListener(new ActionListener(){
          public void actionPerformed(ActionEvent e){
            System.out.println("Checked: " + checkbox.getState());
            System.out.println("Label: " + checkbox.getLabel());
            }
          });
      }
      public static void main(String ap[]){
        new CheckBoxExample();
      }
      }
